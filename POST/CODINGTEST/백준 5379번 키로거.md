```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int count;
    cin >> count;
    for (int i = 0; i < count; i++)
    {
        string s;
        cin >> s;
        list<char> str;
        auto cur = str.end();
        for (char c : s)
        {
            if (c == '<')
            {
                if(cur != str.begin()) cur--;
            }
            else if (c == '>')
            {
                if(cur != str.end()) cur++;
            }
            else if (c == '-')
            {
                if(cur == str.begin()) continue;
                cur = str.erase(--cur);
            }
            else
            {
                str.insert(cur, c);
            }
        }
        for(auto c : str) cout << c;
        cout << '\n';
        
    }
    return 0;
}
```
