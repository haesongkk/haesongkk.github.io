```cpp
#include <bits/stdc++.h>
using namespace std;

stack<int> myStack;

int main() {
    int n;
    cin >> n;
    for(int i = 0; i < n; i++)
    {
        string cmd;
        cin >> cmd;
        if(cmd == "push")
        {
            int num;
            cin >> num;
            myStack.push(num);
        }
        else if(cmd == "pop")
        {
            if(myStack.empty()) cout << "-1" << '\n';
            else 
            {
                cout << myStack.top() << '\n';
                myStack.pop();
            }
        }
        else if(cmd == "size")
        {
            cout << myStack.size() << '\n';
        }
        else if(cmd == "empty")
        {
            cout << myStack.empty() << '\n';
        }
        else if(cmd == "top")
        {
            if(myStack.empty()) cout << "-1" << '\n';
            else cout << myStack.top() << '\n';
        }
    }
}
```
