```cpp
#include <bits/stdc++.h>
using namespace std;

int mul = 1;
int cnt[10];

int main() {
    int tmp;
    for(int i = 0; i < 3; i++) {
        cin >> tmp;
        mul *= tmp;
    }

    while(mul) {
        cnt[mul % 10]++;
        mul /= 10;
    }
    
    for(int c: cnt) cout << c << '\n';
}
```
