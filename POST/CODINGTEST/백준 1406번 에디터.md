[https://www.acmicpc.net/problem/1406](https://www.acmicpc.net/problem/1406)
```cpp
#include <bits/stdc++.h>
using namespace std;

list<char> str;
int cnt;

int main() {
    string s;
    cin >> s;
    cin >> cnt;
    
    for (int i = 0; i < s.length(); i++)
    {
        str.push_back(s[i]);
    }

    char tmp;
    auto curs = str.end();
    for(int i = 0; i < cnt; i++)
    {
        cin >> tmp;
        if(tmp == 'L')
        {
            if (curs != str.begin()) curs--;
        }
        else if(tmp == 'D')
        {
            if (curs != str.end()) curs++;
        }
        else if (tmp == 'B')
        {
            if (curs == str.begin()) continue;
            curs = str.erase(--curs);
        }
        else if (tmp == 'P')
        {
            cin >> tmp;
            str.insert(curs, tmp);
        }
    }

    for(auto it : str) 
    {
        cout << it;
    }
    
}
```
