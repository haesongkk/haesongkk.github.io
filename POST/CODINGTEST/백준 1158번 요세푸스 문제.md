[https://www.acmicpc.net/problem/1158](https://www.acmicpc.net/problem/1158)
```cpp
#include <bits/stdc++.h>
using namespace std;

list<int> peoples;
int main() {
    int n;
    int k;
    cin >> n;
    cin >> k;
    
    for(int i = 0; i < n; i++)
            peoples.push_back(i + 1);
    
    auto it = peoples.begin();
    cout << '<';
    while(!peoples.empty()) 
    {
        for(int i = 0; i < k - 1; i++)
            if(++it == peoples.end()) 
                it = peoples.begin();

        int rt = *it;
        it = peoples.erase(it);
        if(it == peoples.end()) it = peoples.begin();
        
        if(peoples.empty()) cout << rt;
        else cout << rt << ", ";
    }
    cout << '>';
    
    return 0;
}
```
