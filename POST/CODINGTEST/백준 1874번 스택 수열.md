```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    stack<int> st;
    string output;
    int index = 0;
    for(int i = 0; i < n; i++)
    {
        int x;
        cin >> x;
        bool bEnd = false;
        while(true)
        {
            int top = st.empty() ? 0 : st.top();
            if(top > x)
            {
                output = "NO";
                bEnd = true;
                break;
            }
            else if (top < x)
            {
                st.push(++index);
                output += '+';
                output += '\n';
            }
            else
            {
                st.pop();
                output += '-';
                output += '\n';
                break;
            }

            if(index > n)
            {
                output = "NO";
                bEnd = true;
                break;
            }
        }
        
        if(bEnd) break;
    }
    cout << output;
    
    return 0;
}
```
