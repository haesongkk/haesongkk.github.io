[https://www.acmicpc.net/problem/3273](https://www.acmicpc.net/problem/3273)

```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
int* arr;
int x;
int cnt;

int main() {
    cin >> n;
    arr = new int[n];
    for (int i = 0; i < n; i++) cin >> arr[i];
    cin >> x;

    sort(arr, arr + n);

    int* left = arr;
    int* right = &arr[n-1];

    while (left < right) {
        int sum = *left + *right;
        if (sum > x) {
            right--;
        } else if (sum < x) {
            left++;
        } else {
            left++;
            right--;
            cnt++;
        }
    }
    
    cout << cnt;
}
```
