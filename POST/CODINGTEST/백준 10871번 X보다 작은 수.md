```cpp
#include <bits/stdc++.h>
using namespace std;

vector<int> findSmaller(int size, int x, vector<int>& arr) {
    vector<int> smaller = {};
    for(int i = 0; i < size; i++) {
        if(arr[i] < x) {
            smaller.push_back(arr[i]);
        }
    }
    return smaller;
}

int main() {
    int size;
    cin >> size;

    int x;
    cin >> x;

    vector<int> arr;
    for(int i = 0; i < size; i++) {
        int n;
        cin >> n;
        arr.push_back(n);
    }

    vector<int> rt = findSmaller(size, x, arr);
    for(int i: rt) cout << i << ' ';

    return 0;
}
```
