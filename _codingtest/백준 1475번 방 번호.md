```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
int cnt[10];
int rt;

int main() {
    cin >> n;
    
    while (n) {
        cnt[n % 10]++;
        n /= 10;
    }

    cnt[6] += cnt[9];
    cnt[6] = cnt[6] / 2 + cnt[6] % 2;
    cnt[9] = cnt[6];

    for(int i = 0; i < 9; i++) 
        if(cnt[i] > rt) rt = cnt[i];

    cout << rt;
}
```
